### 前言

###  工厂模式
```
    /**
     * 1.工厂模式
     */
    function Person(name) {
        var o = new Object();
        o.name = name;
        o.getName = function () {
            console.log(this.name);
        }
        return o;
    }
    var p1 = new Person('zhangsan');
    p1.getName();
```

### 构造函数模式
```
    /**
     * 2.构造函数模式
     */
    function Person(name) {
        this.name = name;
        this.getName = function () {
            console.log(this.name);
        }
        return this; //默认加上
    }
    var p1 = new Person('zhangsan');
    p1.getName();
```

### 原型模式
```
    /**
     * 3.原型模式
     */
    function Person(name) {
    }
    Person.prototype = {
        contructor: Person,
        name: 'zhangsan',
        getName: function () {
            console.log(this.name);
        }
    }
    var p1 = new Person('zhangsan');
    p1.getName();
```

### 组合模式(构造函数+原型) 推荐
```
    /**
     * 4.组合模式(构造函数+原型) 推荐
     * 优点：
     *  解决了原型模式对于引用对象的缺点
        解决了原型模式没有办法传递参数的缺点
        解决了构造函数模式不能共享方法的缺点
     */
    function Person(name) {
        this.name = name;
    }
    Person.prototype = {
        constructor: Person,
        getName: function () {
            console.log(this.name);
        }
    }
    var p1 = new Person('zhangsan');
    p1.getName();
```

### 动态原型模式
```
    /**
     * 5.动态原型模式
     */
    function Person(name) {
        this.name = name;
        if (typeof this.getName !== 'function') {
            Person.prototype.getName = function () {
                console.log(this.name);
            }
        }
    }
    var p1 = new Person('zhangsan');
    p1.getName();
```

### 寄生构造函数模式
```
    /**
     * 6.寄生构造函数模式

    function Person(name) {
        var o = new Object();

        o.getName = function () {

```
